一些拿 fsharp 随便谢谢的东西。

```fsharp
let rec gcd a b = if b = 0 then a else gcd b (a % b)

let lcm a b = a / gcd a b * b

let rec exgcd a b =
    match b with
    | 0 -> a, 1, 0
    | _ -> let g, x, y = exgcd b (a % b) in g, y, x - a / b * y

let pow =
    let rec loop acc a x =
        match x with
        | 0 -> acc
        | _ when x % 2 = 1 -> loop (acc * a) (a * a) (x / 2)
        | _ -> loop acc (a * a) (x / 2)

    loop 1

let modinv a m =
    let g, x, _ = exgcd a m in

    match g with
    | 1 -> (x % m + m) % m
    | _ -> failwith "No inverse exists"


let fact =
    let rec loop acc n =
        match n with
        | 0 -> acc
        | _ -> loop (n * acc) (n - 1)

    loop 1


let perm n m =
    let rec loop acc n m =
        match m with
        | 0 -> acc
        | _ -> loop (acc * n) (n - 1) (m - 1)

    loop 1 n m


let comb n m =
    let rec loop acc n m =
        match m with
        | 0 -> acc
        | _ -> loop (acc * n) (n - 1) (m - 1)

    loop 1 n m / fact m
```
